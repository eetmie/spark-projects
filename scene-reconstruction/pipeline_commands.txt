# Scene Reconstruction Pipeline - DGX Spark
# iPhone video -> ffmpeg frames -> COLMAP GUI -> 3DGRUT raw.ply -> SuperSplat cleaned.ply -> USDZ
# Run from: /home/masi-pgx/spark-projects/scene-reconstruction

# 1. Create a scene folder and copy the iPhone video into it.
mkdir -p /path/to/my_scene
cp /path/to/video.MOV /path/to/my_scene/

# 2. Extract frames with ffmpeg. This creates only images/.
python tools/extract_video_frames.py /path/to/my_scene/video.MOV

# Optional smaller extraction for faster tests:
# python tools/extract_video_frames.py /path/to/my_scene/video.MOV --frames 150 --max-width 1920 --clear

# Optional blur-aware extraction for handheld/iPhone footage:
# python tools/extract_video_frames.py /path/to/my_scene/video.MOV --frames 200 --select sharp --clear
# python tools/extract_video_frames.py /path/to/my_scene/video.MOV --frames 200 --select sharp --candidate-step 2 --clear
# Sharp mode scores candidates with SIFT + Laplacian and keeps the best frame per time bucket.

# 3. Run COLMAP in the GUI.
python tools/colmap.py /path/to/my_scene

# In COLMAP:
#   File > New project
#     Database : /data/database.db
#     Images   : /data/images
#
#   Processing > Feature extraction
#     Camera model : OPENCV          # iPhone 1x / normal phone video
#                    OPENCV_FISHEYE  # iPhone 0.5x / fisheye / action cam
#     Single camera for all images: enabled
#
#   Processing > Sequential matching
#     Loop detection: off unless the video revisits the same area
#
#   Reconstruction > Start reconstruction
#   File > Save project
#   File > Quit

# Headless alternative, if you do not want the GUI:
# python tools/colmap.py /path/to/my_scene --headless
# python tools/colmap.py /path/to/my_scene --headless --camera-model OPENCV_FISHEYE

# 4. Train 3DGRUT. Default is 3DGRT on DGX Spark.
python tools/train.py /path/to/my_scene

# Useful training options:
# python tools/train.py /path/to/my_scene --iterations 30000
# python tools/train.py /path/to/my_scene --viser
# python tools/train.py /path/to/my_scene --method 3dgut

# 5. Export the direct 3DGRUT PLY.
python tools/export_scene.py /path/to/my_scene
# Output: /path/to/my_scene/raw.ply

# 6. Open raw.ply in SuperSplat.
#    Trim/clean/compress, then export as:
#    /path/to/my_scene/cleaned.ply
#
# Important: keep SH degree at 3 in SuperSplat export.

# 7. Convert the cleaned/compressed SuperSplat PLY to USDZ.
python tools/usd_convert.py /path/to/my_scene/cleaned.ply /path/to/my_scene/scene.usdz
